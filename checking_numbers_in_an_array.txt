array = list(map(int, input("Enter your numbers seperated by spaces").split()))
number= int(input("What number do you want to find?"))
count= 0
length = len(array) - 1
while length >= 0:
    i = (array[length])
    if i == number:
        count += 1
        length -= 1
    else:
        length -= 1
print ("the number",number, "appears", count, "times.")
