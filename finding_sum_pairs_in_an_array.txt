array= [1,3,4,5, 5, 6,8,10]
target = 10
left = 0
right = (len(array))-1
found = 0


while left < right:
    s = array[left]+array[right]
    if s == target:
        print("Pair found", array[left],",", array[right])
        found = 1
        left += 1
        right -=1
    elif s > target:
        right -= 1
    elif s < target:
        left += 1
if found == 0:
    print ("Pair not found")
